153template <typename Element, typename Size>
154inline OperationResult collection2::Tree<Element, Size>::appendChild(
155 TreeNode<Element, Size>* parent,
156 const Element& target,
157 const TreeNodeSide side,
158 TreeNode<Element, Size>** addedNodePtr) {
159 // parentがnullであってはならない
160 if (parent == nullptr) {
161 return OperationResult::Empty;
162 }
163
164 // 新しいノードをもらってくる
165 auto* newNode = getNewNode();
166 if (newNode == nullptr) {
167 if (addedNodePtr != nullptr) {
168 *addedNodePtr = nullptr;
169 }
170 return OperationResult::Overflow;
171 }
172
173 // 追加したい方に値があるなら戻る
174 auto** checkside = &((side == TreeNodeSide::Left) ? parent->lhs : parent->rhs);
175 if (*checkside != nullptr) {
176 return OperationResult::Overflow;
177 }
178
179 // 値をセット
180 newNode->element = target;
181 *checkside = newNode;
182
183 // 新規生成したノードへのポインタを渡す
184 if (addedNodePtr != nullptr) {
185 *addedNodePtr = newNode;
186 }
187 return OperationResult::Success;
188}

190template <typename Element, typename Size>
191inline void collection2::Tree<Element, Size>::removeChild(TreeNode<Element, Size>* target) {
192 // ルートノードは削除できない
193 if (target == rootNode) {
194 return;
195 }
196
197 // リーフならデアクティベートして終わり
198 if (target->isLeaf()) {
199 target->isEnabled = false;
200 return;
201 }
202
203 // そうでなければ再帰的に削除
204 if (target->lhs != nullptr) {
205 removeChild(target->lhs);
206 target->lhs = nullptr;
207 }
208 if (target->rhs != nullptr) {
209 removeChild(target->rhs);
210 target->rhs = nullptr;
211 }
212}

216 for (Size i = 0; i < internalDataSize; i++) {
217 if (internalData[i].isEnabled) {

220
221 // 見つかったらノードを初期化して返す
222 internalData[i].isEnabled = true;
223 internalData[i].lhs = nullptr;
224 internalData[i].rhs = nullptr;
225 return &(internalData[i]);
226 }
227 return nullptr;
