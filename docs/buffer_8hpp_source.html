118template <typename Element>
119Buffer<Element>::Buffer(Element* const data, const buffer_size_t& dataSize) : internalData(data) {
120 // ゼロ長のバッファなら何もしない
121 if (dataSize == 0) {
122 internalDataSize = dataSize;
123 return;
124 }
125
126 // 与えられたサイズを上回らない最大の2の冪数を探す
127 const uint8_t BufferSizeBitLength = sizeof(buffer_size_t) * CHAR_BIT; // バッファサイズのビット数
128 uint8_t currentBitPosition = BufferSizeBitLength;
129 buffer_size_t candidate = 0;
130 do {
131 candidate = static_cast<buffer_size_t>(1 << currentBitPosition);
132 if ((candidate & dataSize) != 0) {
133 break;
134 }
135 } while (currentBitPosition--);
136 internalDataSize = candidate;
137};

139template <typename Element>
140OperationResult Buffer<Element>::append(const Element& data) {
141 // サイズ0のバッファに値を追加することはできない
142 if (internalDataSize == 0) {
143 return OperationResult::Overflow;
144 }
145
146 // バッファがいっぱいなら、要素を一つ読み捨てる
147 if (!hasSpace()) {
148 auto result = pop(nullptr);
149 }
150
151 // tailの位置にデータを書き込む
152 *(internalData + tail) = data;
153
154 tail = (tail + 1) & (internalDataSize - 1);
155 count++;
156
157 return OperationResult::Success;

162 // バッファが空なら戻る

167 // 読み出して渡す。nullptrなら何もしないでheadを進める。

170 }

172 head = (head + 1) & (internalDataSize - 1);
173 count--;

175 return OperationResult::Success;
