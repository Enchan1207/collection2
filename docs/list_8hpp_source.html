191template <typename Element>
192OperationResult List<Element>::append(const Element& element) {
193 // 新しいノードを取得し、値を設定
194 auto* newNode = getNewNode();
195 if (newNode == nullptr) {
196 return OperationResult::Overflow;
197 }
198 newNode->element = element;
199
200 // 先頭の値がないならそこにセットして戻る
201 if (headPtr == nullptr) {
202 headPtr = newNode;
203 tailPtr = newNode;
204 count++;
205 return OperationResult::Success;
206 }
207
208 // tailに接続
209 tailPtr->next = newNode;
210 newNode->previous = tailPtr;
211 tailPtr = newNode;
212 count++;
213
214 return OperationResult::Success;

217template <typename Element>
218OperationResult List<Element>::insert(const list_size_t& index, const Element& element) {
219 // 新しいノードを取得し、値を設定
220 auto* newNode = getNewNode();
221 if (newNode == nullptr) {
222 return OperationResult::Overflow;
223 }
224 newNode->element = element;
225
226 // 先頭の値がないならそこにセットして戻る
227 if (headPtr == nullptr) {
228 headPtr = newNode;
229 tailPtr = newNode;
230 count++;
231 return OperationResult::Success;
232 }
233
234 // 先頭への追加の場合、現在のheadが新しいノードのnextとなる
235 if (index == 0) {
236 newNode->next = headPtr;
237 headPtr = newNode;
238 count++;
239 return OperationResult::Success;
240 }
241
242 // 先頭から最大index回nextを辿り、追加位置の直前のノードを取得
243 auto* previousNode = headPtr;
244 list_size_t currentIndex = 0;
245 while (previousNode->next != nullptr && currentIndex < index - 1) {
246 previousNode = previousNode->next;
247 currentIndex++;
248 }
249
250 // 接続
251 if (previousNode->next == nullptr) {
252 // 前のノードの次がnull -> リスト終端への追加
253 tailPtr = newNode;
254 } else {
255 // 前のノードの次が存在 -> リスト途中への追加
256 previousNode->next->previous = newNode;
257 }
258 newNode->next = previousNode->next;
259
260 newNode->previous = previousNode;
261 previousNode->next = newNode;
262
263 count++;
264
265 return OperationResult::Success;
266}

268template <typename Element>
269OperationResult List<Element>::pop(Element* const element) {
270 if (tailPtr == nullptr) {
271 return OperationResult::Empty;
272 }
273
274 // 対象ノードはtailから参照できる
275 auto* targetNode = tailPtr;
276
277 // tailの位置に格納されている情報を渡す
278 if (element != nullptr) {
279 *element = targetNode->element;
280 }
281
282 // tailを一つ戻す
283 tailPtr = targetNode->previous;
284 if (tailPtr == nullptr) {
285 // tailがnullになった=リストが空になった
286 headPtr = nullptr;
287 }
288
289 // ノードを無効化する
290 targetNode->isEnabled = false;
291 targetNode->next = nullptr;
292 targetNode->previous = nullptr;
293
294 count--;
295
296 return OperationResult::Success;
297}

299template <typename Element>
300OperationResult List<Element>::remove(const list_size_t& index, Element* const element) {
301 if (headPtr == nullptr) {
302 return OperationResult::Empty;
303 }
304
305 // 先頭のノードを削除する場合
306 if (index == 0) {
307 // 先頭を一つ先にずらす
308 auto* targetNode = headPtr;
309 headPtr = targetNode->next;
310 if (headPtr == nullptr) {
311 // headがnullになった=リストが空になった
312 tailPtr = nullptr;
313 }
314
315 // ノードに格納されている情報を渡す
316 if (element != nullptr) {
317 *element = targetNode->element;
318 }
319
320 // 要素を無効化
321 targetNode->isEnabled = false;
322 targetNode->next = nullptr;
323 targetNode->previous = nullptr;
324
325 count--;
326
327 return OperationResult::Success;
328 }
329
330 // 先頭からindex回、最大で配列の末尾までnextを辿り、削除位置のノードを取得
331 auto* targetNode = headPtr;
332 list_size_t currentIndex = 0;
333 while (targetNode->next != nullptr && currentIndex < index) {
334 targetNode = targetNode->next;
335 currentIndex++;
336 }
337
338 // ノードに格納されている情報を渡す
339 if (element != nullptr) {
340 *element = targetNode->element;
341 }
342
343 // 前後を再接続
344 if (targetNode->next == nullptr) {
345 // 前のノードの次がnull -> リスト終端の削除
346 tailPtr = targetNode->previous;
347 } else {
348 // 前のノードの次が存在 -> リスト中間の削除
349 targetNode->next->previous = targetNode->previous;
350 }
351 targetNode->previous->next = targetNode->next;
352
353 // ノードを無効化
354 targetNode->isEnabled = false;
355 targetNode->next = nullptr;
356 targetNode->previous = nullptr;
357
358 count--;
359
360 return OperationResult::Success;
