156OperationResult List<Element>::append(const Element& element) {
157 // 新しいノードを取得し、値を設定
158 auto* newNode = getNewNode();
159 if (newNode == nullptr) {

179template <typename Element>
180OperationResult List<Element>::insert(const list_size_t& index, const Element& element) {
181 // 新しいノードを取得し、値を設定
182 auto* newNode = getNewNode();
183 if (newNode == nullptr) {
184 return OperationResult::Overflow;
185 }
186 newNode->element = element;
187
188 // 先頭の値がないならそこにセットして戻る
189 if (head == nullptr) {
190 head = newNode;
191 tail = newNode;
192 return OperationResult::Success;
193 }
194
195 // 先頭への追加の場合、現在のheadが新しいノードのnextとなる
196 if (index == 0) {
197 newNode->next = head;
198 head = newNode;
199 return OperationResult::Success;
200 }
201
202 // 先頭から最大index回nextを辿り、追加位置の直前のノードを取得
203 auto* previousNode = head;
204 list_size_t currentIndex = 0;
205 while (previousNode->next != nullptr && currentIndex < index - 1) {
206 previousNode = previousNode->next;
207 currentIndex++;
208 }
209
210 // 接続
211 if (previousNode->next == nullptr) {
212 // 前のノードの次がnull -> リスト終端への追加
213 tail = newNode;
214 } else {
215 // 前のノードの次が存在 -> リスト途中への追加
216 previousNode->next->previous = newNode;
217 }
218 newNode->next = previousNode->next;
219
220 newNode->previous = previousNode;
221 previousNode->next = newNode;
222
223 return OperationResult::Success;
224}

226template <typename Element>
227OperationResult List<Element>::pop(Element* const element) {
228 if (tail == nullptr) {
229 return OperationResult::Empty;
230 }
231
232 // 対象ノードはtailから参照できる
233 auto* targetNode = tail;
234
235 // tailの位置に格納されている情報を渡す
236 if (element != nullptr) {
237 *element = targetNode->element;
238 }
239
240 // tailを一つ戻す
241 tail = targetNode->previous;
242 if (tail == nullptr) {
243 // tailがnullになった=リストが空になった
244 head = nullptr;
245 }
246
247 // ノードを無効化する
248 targetNode->isEnabled = false;
249 targetNode->next = nullptr;
250 targetNode->previous = nullptr;
251
252 return OperationResult::Success;

255template <typename Element>
256OperationResult List<Element>::remove(const list_size_t& index, Element* const element) {
257 if (head == nullptr) {
258 return OperationResult::Empty;
259 }
260
261 // 先頭のノードを削除する場合
262 if (index == 0) {
263 // 先頭を一つ先にずらす
264 auto* targetNode = head;
265 head = targetNode->next;
266 if (head == nullptr) {
267 // headがnullになった=リストが空になった
268 tail = nullptr;
269 }
270
271 // ノードに格納されている情報を渡す
272 if (element != nullptr) {
273 *element = targetNode->element;
274 }
275
276 // 要素を無効化
277 targetNode->isEnabled = false;
278 targetNode->next = nullptr;
279 targetNode->previous = nullptr;
280
281 return OperationResult::Success;
282 }

284 // 先頭からindex回、最大で配列の末尾までnextを辿り、削除位置のノードを取得
285 auto* targetNode = head;
286 list_size_t currentIndex = 0;
287 while (targetNode->next != nullptr && currentIndex < index) {
288 targetNode = targetNode->next;
289 currentIndex++;
290 }
291
292 // ノードに格納されている情報を渡す
293 if (element != nullptr) {
294 *element = targetNode->element;
295 }
296
297 // 前後を再接続
298 if (targetNode->next == nullptr) {
299 // 前のノードの次がnull -> リスト終端の削除
300 tail = targetNode->previous;
301 } else {
302 // 前のノードの次が存在 -> リスト中間の削除
303 targetNode->next->previous = targetNode->previous;
304 }
305 targetNode->previous->next = targetNode->next;
306
307 // ノードを無効化
308 targetNode->isEnabled = false;
309 targetNode->next = nullptr;
310 targetNode->previous = nullptr;
311
312 return OperationResult::Success;
