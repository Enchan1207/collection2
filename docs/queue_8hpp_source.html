121template <typename Element>
122Queue<Element>::Queue(Element* const data, const queue_size_t& dataSize) : internalData(data) {
123 // ゼロ長のキューなら何もしない
124 if (dataSize == 0) {
125 internalDataSize = dataSize;
126 return;
127 }
128
129 // 与えられたサイズを上回らない最大の2の冪数を探す
130 const uint8_t queueSizeBitLength = sizeof(queue_size_t) * CHAR_BIT; // キューサイズのビット数
131 uint8_t currentBitPosition = queueSizeBitLength;
132 queue_size_t candidate = 0;
133 do {
134 candidate = static_cast<queue_size_t>(1 << currentBitPosition);
135 if ((candidate & dataSize) != 0) {
136 break;
137 }
138 } while (currentBitPosition--);
139 internalDataSize = candidate;
140};

143OperationResult Queue<Element>::enqueue(const Element& data) {
144 // キューがいっぱいなら戻る

152 tail = (tail + 1) & (internalDataSize - 1);
153 count++;
