100template <typename Element>
101Queue<Element>::Queue(Element* const data, const queue_size_t& dataSize) : internalData(data) {
102 // ゼロ長のキューなら何もしない
103 if (dataSize == 0) {
104 internalDataSize = dataSize;
105 return;
106 }
107
108 // 与えられたサイズを上回らない最大の2の冪数を探す
109 const uint8_t queueSizeBitLength = sizeof(queue_size_t) * CHAR_BIT; // キューサイズのビット数
110 uint8_t currentBitPosition = queueSizeBitLength;
111 queue_size_t candidate = 0;
112 do {
113 candidate = static_cast<queue_size_t>(1 << currentBitPosition);
114 if ((candidate & dataSize) != 0) {
115 break;
116 }
117 } while (currentBitPosition--);
118 internalDataSize = candidate;
119};
